\subsection*{Differentialkvotient for $f(x)+g(x)$}
\begin{align*}
\intertext{\hspace{1em} \circnum{1} Opskriv sekanthældningen:}
\frac{\Delta y}{\Delta x}&=\frac{\left(f(x+h)+g(x+h)\right)-\left(f(x)+g(x)\right)}{h}\\
\intertext{\hspace{1em} \circnum{2} Reducér udtrykket:}
&=\frac{f(x+h)+g(x+h)-f(x)-g(x)}{h}\\
&=\frac{f(x+h)-f(x)+g(x+h)-g(x)}{h}\\
&=\frac{f(x+h)-f(x)}{h}+\frac{g(x+h)-g(x)}{h}\\
\intertext{\hspace{1em} \circnum{3} Lad $h\rightarrow 0$:}
f'(x)&=\lim_{h\rightarrow 0}\left(\frac{f(x+h)-f(x)}{h}+\frac{g(x+h)-g(x)}{h}\right)\\
&=\lim_{h\rightarrow 0}\left(\frac{f(x+h)-f(x)}{h}\right)+\lim_{h\rightarrow 0}\left(\frac{g(x+h)-g(x)}{h}\right)\\
&=f'(x)+g'(x)
\end{align*}
