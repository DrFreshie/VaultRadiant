    f(x) = a \cdot x^{2} + b \cdot x + c.

\begin{enumerate}
    \item Bestem fortegnet for hvert af tallene $a$ og $c$. Begrund dit svar.

\subsection{}
\begin{enumerate}
    \item Forklar linje for linje nedenstående reduktion af et udtryk.
\end{enumerate}

\begin{align*}
&&	&						&&\underset{\rule{0.8\linewidth}{0pt}}{\textbf{Forklaring:}}\\[1em]
&&	&\frac{(x+2)^2 - 4}{x} 		&&\underset{\rule{0.8\linewidth}{0.4pt}}{\text{Udtrykket skrives op.}}\\[2em]
&&	=&\ \frac{x^2 + 4 + 4x - 4}{x}	&&\underset{\rule{0.8\linewidth}{0.4pt}}{\text{}}\\[2em]
&&	=&\ \frac{x^2 + 4x}{x}			&&\underset{\rule{0.8\linewidth}{0.4pt}}{\text{}}\\[2em]
&&	=&\ x + 4					&&\underset{\rule{0.8\linewidth}{0.4pt}}{\text{}}\\[2em]
\end{align*}

\[
    g(x) = f(x) + 2.
\]

\begin{enumerate}

    \item Har $f$ og $g$ lige mange nulpunkter? Begrund dit svar.

\[
    f(x) = x^2 + 2x - 3.
\]

\begin{enumerate}
    \item Bestem diskriminanten $d$, og bestem andengradspolynomiets rødder.
    \item Tegn en skitse af grafen for $f$ i hånden.
\end{enumerate}

\begin{enumerate}
    \item Gør for hvert af tallene $a$ og $c$ rede for, om det er positivt eller negativt.
