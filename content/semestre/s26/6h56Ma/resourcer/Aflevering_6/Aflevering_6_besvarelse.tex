\subsection{}
En funktion $f$ er givet ved
$$
f(x) = x^2 - 2x .
$$

\begin{enumerate}
    \item Bestem $f(5)$.
\end{enumerate}
\svar{
	$$f(5)=5^2-2\cdot 5=20$$
}

\begin{enumerate}[resume]
    \item Bestem $f'(5)$.
\end{enumerate}
\svar{
	\begin{align*}
	f'(x)&=2x-2\\
	f'(5)&=2\cdot 5-2=8
	\end{align*}
}

\subsection{}
En funktion $f$ er givet ved
$$
f(x)=6\cdot\sqrt{x}\, .
$$

\begin{enumerate}
    \item Bestem $f'(x)$.
\end{enumerate}
\svar{
	Vi bruger konstantgangeregelen:
	$$f'(x)=6\cdot \frac{1}{2\cdot \sqrt{x}}=\frac{3}{\sqrt{x}}$$
}

En anden funktion $g$ er givet ved
$$
g(x)=k\cdot\sqrt{x}, \quad \text{hvor } k \text{ er et tal}.
$$
Det oplyses, at
$$
g'(4)=3 .
$$

\begin{enumerate}
    \setcounter{enumi}{1}
    \item Bestem tallet $k$.
\end{enumerate}
\svar{
	Bruger vi konstantgangeregelen med $k$ får vi
	$$f'(x)=k\cdot \frac{1}{2\cdot \sqrt{x}}$$
	med $x=4$ skal dette udtryk give 3, og vi har altså ligningen:
	\begin{align*}
		k\cdot \frac{1}{2\cdot \sqrt{4}}&=3&&\Leftrightarrow\\
		\frac{k}{4}&=3&&\Leftrightarrow\\
		k&=3\cdot 4 = 12
	\end{align*}
}

Figuren viser grafen for en differentiabel funktion $f$ samt tangenten til grafen i punktet $P(1,4)$.

Det oplyses, at $f'(1)=0{,}5$.

\begin{enumerate}
  \item Bestem en ligning for tangenten til grafen for $f$ i punktet $P(1,4)$.
\end{enumerate}
\svar{
	Da $f'(1)=0{,}5$ er hældningen i $x=1$ altså $a=0{,}5$. $b$-værdien kan vi finde med punktet og formlen $b=y_1-a\cdot x_1$:
	$$b=4-0{,}5\cdot 1=3{,}5$$
	Ligningen for tangenten til grafen for $f$ i punktet $P(1,4)$ bliver derfor
	$$y=0{,}5x+3{,}5$$
}

\subsection{}
En funktion er givet ved
$$
f(x) = x^3 \cdot \ln(x).
$$

\begin{enumerate}
    \item Bestem $f(1)$.
\end{enumerate}
\svar{
	$\ln(1)$ er svaret på hvad e skal opløftes i for at få $1$, så $\ln(1)=0$, og vi får:
	$$f(1)=1^3\cdot \ln(1)=0$$
}

\begin{enumerate}[resume]
    \item Bestem $f'(x)$.
\end{enumerate}
\svar{
	Vi bruger produktregelen:
	$$f'(x)=3x^2\cdot \ln(x)+x^3\cdot \tfrac1x=3x^2\cdot\ln(x)+x^2=x^2(3\ln(x)+1)$$
}

\subsection{}
En funktion $f$ er givet ved
\[
    f(x) = 5\cdot\ln(x) + 4x .
\]

\begin{enumerate}
    \item Bestem en ligning for tangenten til grafen for $f$ i punktet med første-koordinat $x=1$.
\end{enumerate}
\svar{
	Først kan vi med CAS udregne $y$-værdien for punktet med $x=1$:
	$$y=f(1)=4$$
	
	Vi kan dernæst bruge CAS til at udregne tangentens hældning $f'(1)$:
	$$a=f'(1)=9$$

	Herefter kan vi bruge formlen for $b$-værdien:
	$$b=y-ax=4-9\cdot 1=-5$$

	I alt bliver ligningen for tangent til grafen for $f$ i punktet med $x=1$:
	$$y=9x-5$$
}	

\subsection{}
En funktion $V$ har forskriften
$$
V(x)=-2\cdot x^{3}+10\cdot x^{2}, \qquad 0<x<5.
$$

\begin{enumerate}
    \item Bestem maksimum for $V$ ved brug af $V'(x)$.
\end{enumerate}
\svar{
	Vi løser ligningen med CAS:
	\begin{align*}
		V'(x)&=0&&\Leftrightarrow\\
		x&=0\quad V \quad x=\frac{10}{3}
	\end{align*}

	Da $0<x$ må maksimum være i $x=\frac{10}{3}$.
}

\begin{enumerate}   
	\item Argumentér for, at forskriften $V(x)$ beskriver kassens rumfang som funktion af $x$.
\end{enumerate}
\svar{

	$$x\cdot x\cdot (10-2x)$$

	$$x\cdot x\cdot (10-2x)=x^2(10-2x)=x^2\cdot 10-x^2\cdot 2x=10\cdot x^2-2\cdot x^3$$
}
